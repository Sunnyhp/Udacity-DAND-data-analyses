# Across all columns
df = pd.DataFrame({
        'col_0': [0, 0, 1, 1],
        'col_1': [0, 0, 2, 3]
    })
print(df)
df_deduplicated = df.drop_duplicates() # default keep='first'
print(df_deduplicated)

# last: Drop duplicates except for the last occurrence.
